#!/usr/bin/env python

import numpy as np
import matplotlib.pyplot as plot

time = np.arange(0,10,0.1);
amplitude = np.floor(time)
plot.plot(time,amplitude)

plot.title("Floor function")
plot.xlabel("Time")
plot.ylabel("Amplitude = floor(time)")
plot.grid(True, which='both')
plot.axhline(y = 0, color='k')
plot.show()
